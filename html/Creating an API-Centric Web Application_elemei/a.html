                                                                                                                                                      For anyone who's having trouble getting information from the API server. The cURL needs a small change in code so the CURLOPT_POSTFIELDS can be send to the server.

                                                                                                                                                      curl_setopt($ch, CURLOPT_POSTFIELDS, http_build_query($params));

                                                                                                                                                      If the API server uses SSL you can setup the options below, or just put the SSL cert info in the option.

                                                                                                                                                      curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
                                                                                                                                                      curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, false);
